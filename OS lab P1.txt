#include<stdio.h>
#include<unistd.h>
#include<sys/types.h>
#include<sys/stat.h>
#include<fcntl.h>
int main()
{
int n, fd1, fd2;
char buff[50]={};
fd1 = open("src.text", O_RDONLY);
fd2 = open("dest.txt", O_WRONLY|O_CREATE);
n = read(fd1, buff, 15);
write(fd2, buff, n);
printf("Contents of dest are :%s\n", buff);
close(fd1);
close(fd2);
return 0;
}
